4) select department_id, count(*) as total_employees

5) select department_id, avg(salary) as avg_salary
   from employees
   group by department_id

9) select month(order_date) as month, count(*) as total_orders
   from orders

16) select c.region, sum(o.total_amount) as total_order_amount
    from orders o
    join customers c on o.customer_id = c.customer_id
    group by c.region;

17) select b.author_id, sum(s.quantity) as total_books_sold
    from books b
    join sales s on b.book_id = s.book_id
    group by b.author_id
    having sum(s.quantity) > 100;
